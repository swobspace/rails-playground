== Person

[source,sh]
----
bin/rails g scaffold Person \
  sn:string \
  givenname:string \
  username:string \
  birthdate:date \
  city:string \
  active:boolean \
  category:belongs_to \
  --no-stylesheets --no-javascripts --no-helper
----
